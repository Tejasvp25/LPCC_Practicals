MACRO INCR &A,&B,&REG=CREG
MOVER &REG,&A
ADDS1 &A,&B
MOVEM &REG,&A
MEND
MACRO ADDS1 &C,&D
MOVER AREG,&C
ADD AREG,&D
MOVEM AREG,&D
WRITE &D
MEND
MACRO SUBS1 &C1,&D1
MOVER BREG,&C1
SUB	BREG,&D1
MOVEM BREG,&D1
WRITE &D1
MEND
START 200
READ N1
READ N2
ADDS1 N1,N2
SUBS1 N1,N2
READ N1
READ N2
INCR N3,N4,DREG
STOP
N1 DS 2
N2 DS 2
N3 DS 2
N4 DS 2
END
